province_event = {
	id = india.1
	title = "india.1.t"
	desc = "india.1.d"
	picture = invasion
	fire_only_once = yes
	major = yes
	
	trigger = {
		area = rajarata_area
		has_port = yes
		is_year = 240
		NOT = { started_in = 260.1.1 }
		NOT = { has_global_flag = aryan_invasion_sri_lanka_flag }
		owner = { 
			NOT = { num_of_cities = 5 }
			is_tribal = yes
		}
	}
	
	mean_time_to_happen = {
		months = 160
		modifier = {
				factor = 0.75
				province_id = 3342
			}
	}
	
	immediate = { 
		set_global_flag = aryan_invasion_sri_lanka_flag
		hidden_effect = {
					cede_province = SMV
					add_core = SMV
					change_culture = sinhala
					change_religion = brahmin
					infantry = SMV
					infantry = SMV
					infantry = SMV
					infantry = SMV
					infantry = SMV
					every_province = {
						limit = { area = rajarata_area }
						add_core = SMV
					}
		}
	}
	
	option = {
		name = "india.1.a"	
				tooltip = { cede_province = SMV }
				declare_war_with_cb = {
					who = SMV
					casus_belli = cb_core
				}
	}
	option = {
		name = "india.1.b"	
				tooltip = { cede_province = SMV }
				
	}

}

	id = india.9
	title = "india.9.t"
	desc = "india.9.d"
	picture = battle
	
	trigger = {
			owner = { has_ruler_flag = ashvamedha_ruler_flag }
			OR = {
				is_capital = yes
				any_neighbor_province = { 	is_capital = yes
											owned_by = ROOT } }
			NOT = { has_province_flag = ashvamedha_ritual_flag }
	}
	
	mean_time_to_happen = {
		months = 9
				modifier = {
					factor = 0.8
					owner = { NOT = { legitimacy = 70 } }
				}
				modifier = {
					factor = 1.2
					owner = { legitimacy = 90 } 
				}
	}
	
	immediate = {
		set_province_flag = ashvamedha_ritual_flag
	}
	
	option = {
		name = "india.9.a"
		ai_chance = { factor = 80 }
			spawn_rebels = {
				type = noble_rebels
				size = 1
			}
			random_list = {
				90 = {
					custom_tooltip = india.9.tta
					owner = { add_tribal_allegiance = 5 }
				}
				10 = {
					custom_tooltip = india.9.ttb
					hidden_effect = {
						country_event = {
							id = india.12
							days = 1
						}
					}
				}
			}
	}
	option = {
		name = "india.9.b"
		ai_chance = { factor = 20 }
