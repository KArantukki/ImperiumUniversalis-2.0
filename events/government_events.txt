	desc = "government.1.d"
	picture = generals
	is_triggered_only = yes

		has_leaders = {
           value = 1
           type = general
           include_monarch = no
           include_heir = no
       }
	}

		days = 1
			modifier = {
				factor = 0.5

			}

		name = "government.1.a"

				type = general
			}
	}
	option = {
		name = "government.1.b"

	}
}

			}

			}

			}
	}

				limit = {
					any_neighbor_province = { NOT = { owned_by = ROOT } }
				}
				every_neighbor_province = {
					limit = { NOT = { owned_by = ROOT } }
					add_claim = ROOT
				}
			}
	}
