country_event = {
	id = sparta.1
	title = "sparta.1.t"
	desc = "sparta.1.d"
	picture = spartans
	is_triggered_only = yes
	
	
	trigger = {
		has_reform = spartan_monarchy_reform
	}
	
	mean_time_to_happen = {
		days = 1
	}
	
	
	# Eurypontid
	option = {
		name = "sparta.1.a"
		trigger = { dynasty = "Agiad" }
		define_heir = {
			dynasty = "Eurypontid"
			age = 25
		}
	}
	option = {
		name = "sparta.1.b"
		trigger = { dynasty = "Eurypontid" }
		define_heir = {		
			dynasty = "Agiad"
			age = 25
		}
	}

}
