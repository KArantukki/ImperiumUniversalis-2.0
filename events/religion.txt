		}
	}
	
	option = {
		name = "religion.4.a"
		random_province = {
				limit = { 
					NOT = { religion = cybele }
				}
				change_religion = cybele
			}
			
	}
}

########################################################################################
# New Shrine
########################################################################################
country_event = {
    id = religion.8
    title = "religion.8.t"
    desc = "religion.8.d"
    picture = religious
            
	is_triggered_only = yes
	
    trigger = {
        uses_patriarch_authority = yes
		treasury = 100
        any_owned_province = {
            num_free_building_slots = 1
            NOT = { has_building = shrine }
            NOT = { has_building = small_temple }
            NOT = { has_building = great_temple }
        }
    }
    
    mean_time_to_happen = {
        days = 1
    }
    
    option = {
        name = "religion.8.a"
        random_owned_province = {
            limit = {
                NOT = {    has_building = shrine }
				NOT = { has_building = small_temple }
				NOT = { has_building = great_temple }
                num_free_building_slots = 1
            }
            add_building = shrine
        }
            add_treasury = -100
            add_patriarch_authority = 0.05
			add_stability = 1
    }
    
    option = {
        name = "religion.8.b"
        add_patriarch_authority = -0.05
        add_stability = -1
    }
}

########################################################################################
# ruler follows new cult
########################################################################################
country_event = {
    id = religion.9
    title = "religion.9.t"
    desc = "religion.9.d"
    picture = religious
            
	is_triggered_only = yes
	
    trigger = {
        OR = {
			has_estate = estate_priests
			uses_patriarch_authority = yes
		}
		NOT = { religious_unity = 0.85 }
    }
    
    mean_time_to_happen = {
        days = 1
    }
    
    option = {
        name = "religion.9.a"
        if = {
			limit = {  uses_patriarch_authority = yes }
			add_patriarch_authority = 0.05
		}
        if = {
			limit = {  has_estate = estate_priests }
			add_estate_loyalty = {
				estate = estate_priests
				loyalty = 5
			}
		}
			add_legitimacy = -5
			add_prestige = -5
    }
    
    option = {
        name = "religion.9.b"
        if = {
			limit = {  uses_patriarch_authority = yes }
			add_patriarch_authority = -0.05
		}
        if = {
			limit = {  has_estate = estate_priests }
			add_estate_loyalty = {
				estate = estate_priests
				loyalty = 5
			}
		}
			add_legitimacy = 5
			add_prestige = 5
    }
}

# religion spreads
########################################################################################
province_event = {
	id = religion.11
	title = "religion.11.t"
	desc = "religion.11.d"
	picture = religious
	hidden = yes
	
	trigger = {
			any_neighbor_province = {
				NOT = { religion = ROOT }
				religion_group = pagan_group 
				owner = { NOT = { patriarch_authority = 0.25 } }
			}
				OR = {
					owner = { patriarch_authority = 0.25 }
					owner = { uses_patriarch_authority = no }
				}
				
			
	}
	
	immediate = {
		
			random_neighbor_province = {
				limit = {
				religion_group = pagan_group 
				owner = { NOT = { patriarch_authority = 0.25 } 
					}
				}
				change_religion = ROOT
				province_event = {
					id = religion.12
					days = 1
				}
			}
	}
	
	mean_time_to_happen = {
		months = 65000
		modifier = {
				factor = 2.0
					owner = { is_tribal = yes }
			}
		modifier = {
				factor = 2.0
					owner = { NOT = { 	patriarch_authority = 0.4 } }
			}
		modifier = {
				factor = 1.5
					owner = { NOT = { 	patriarch_authority = 0.6 } }
			}
		modifier = {
				factor = 0.85
					owner = { 			patriarch_authority = 0.8 }
			}
		modifier = {
				factor = 0.9
				owner = { DIP = 4 }
			}
		modifier = {
				factor = 1.25
				owner = { full_idea_group = syncretic_religious_ideas }
			}
		modifier = {
				factor = 0.85
				owner = { full_idea_group = superior_religious_ideas }
			}
	}

	option = {
		name = "religion.11.a"
	}
}
