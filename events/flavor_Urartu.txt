				factor = 0.5

					stability = -1
				}
			}
			modifier = {
				factor = 0.5
				is_year = 165
			}
			modifier = {
				factor = 0.5
				URT = { 
					OR = {
						DIP = 1
						MIL = 1
					}
				}
			}

			modifier = {
				factor = 1.5
				NOT = { stability = 0 }
			}
			modifier = {
				factor = 1.5
				war_exhaustion = 2
			}
	}
	
	# Attack Urartu
	option = {
		name = "urartu.1.a"
		ai_chance = { factor = 99 }
			set_country_flag = attack_on_urartu_flag
			add_prestige = 5
			add_casus_belli = {
					target = URT
					type = cb_border_war
					months = 120
				}
			if = {
					limit = { ai = yes }
					declare_war_with_cb = {
						who = URT
						casus_belli = cb_border_war
					}
					hidden_effect = {
						URT = {
							add_stability = -1
							}
					}
				}
			}
	# dont attack Urartu
	option = {
		name = "urartu.1.b"
			add_prestige = -10
			set_country_flag = attack_on_urartu_flag
	}
