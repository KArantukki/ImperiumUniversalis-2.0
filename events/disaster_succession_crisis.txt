namespace = succession_crisis_event

########################################################################################
# succession crisis starts
########################################################################################
country_event = {
	id = succession_crisis_event.1	
	title = "succession_crisis_event.1.t"
	desc = "succession_crisis_event.1.d"
	picture = civilwar
	is_triggered_only = yes

	major = yes
	
	trigger = {

	}
	
	
	mean_time_to_happen = {
		days = 1
		
	}

	immediate = {
		hidden_effect = {
		random_owned_province = {
			limit = {
				is_core = ROOT 
			}
			spawn_rebels = {
				type = 	pretender_rebels
				size = 3
				win = yes
				}
			}
		}
	}
	
	option = {
		name = "succession_crisis_event.1.a"
		ai_chance = { factor = 50 }
			add_stability = -1
			if = {
				limit = {
					any_owned_province = { can_spawn_rebels = nationalist_rebels }
				}
				random_owned_province = {
					limit = { can_spawn_rebels = nationalist_rebels }
					spawn_rebels = {
						type = 	nationalist_rebels
						size = 4
						win = yes
					}
				}
			}
	}
	option = {
		name = "succession_crisis_event.1.b"
		ai_chance = { factor = 50 }
			trigger = {
				num_of_cities = 6
				any_owned_province = {
					is_in_capital_area = no
					OR = {
						nationalism = 1
						can_spawn_rebels = nationalist_rebels
					}
				}
			}
			every_owned_province = {
				limit = {
					is_in_capital_area = no
					OR = {
						nationalism = 1
						can_spawn_rebels = nationalist_rebels
					}
				}
				cede_province = previous_owner
			}
			add_legitimacy = 10
			add_yearly_manpower = 0.5
			
	}

}
########################################################################################
