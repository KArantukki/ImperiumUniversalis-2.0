########################################################################################
# Isagoras
########################################################################################
country_event = {
	id = athen.1
	title = "athen.1.t"
	desc = "athen.1.d"
	picture = senat_conflict
	
	
	trigger = {
		tag = ATH
		has_country_modifier = solons_reform
		faction_in_power = fac_oligarchs
		is_year = 240
	}
	
	mean_time_to_happen = {
		months = 80
	}
	
	option = {
		name = "athen.1.a"
			remove_country_modifier = solons_reform
			add_faction_influence = { faction = fac_democrats influence = -10 }
			define_ruler = {
				name = "Isagoras"
				dynasty = "Tisandae"
				mil = 2
				adm = 4
				dip = 2
				fixed = yes
			}
			add_estate_loyalty = {
				estate = estate_citizens
				loyalty = -10
			}
	}
	option = {
		name = "athen.1.b"
			add_faction_influence = { faction = fac_oligarchs influence = -10 }
			add_estate_loyalty = {
				estate = estate_citizens
				loyalty = 5
			}
	}
}

			}

	}
}
