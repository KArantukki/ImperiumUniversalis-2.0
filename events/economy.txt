	title = "economy.2.t"
	desc = "economy.2.d"

	is_triggered_only = yes

		any_owned_province = {
			trade_goods = iron
		}

		name = "economy.2.b"
		ai_chance = { factor = 30 }
			
			add_country_modifier = {
				name = iron_for_armors
				duration = 1825
			}
	}
	option = {
		name = "economy.2.c"
		ai_chance = { factor = 40 }
				add_years_of_income = 0.1
			
	}
}

########################################################################################
# people build streets
########################################################################################
country_event = {
	id = economy.10
	title = "economy.10.t"
	desc = "economy.10.d"
	picture = construction
	is_triggered_only = yes
	
	trigger = {
		any_owned_province = {
			OR = {
				can_build = gravel_road
				can_build = paved_road
				can_build = imperial_road
			}
			has_construction = no
		}
		is_tribal = no
	}
	
	mean_time_to_happen = {
		days = 1
	}
	
	
	option = {
		name = "economy.10.a"
		ai_chance = { factor = 90 }
		random_owned_province = {
			if = {
				limit = { can_build = gravel_road }
				add_dip_power = -20
				add_building_construction = {
					building = gravel_road
					speed = 0.5 # 50% of original speed
					cost = 0
				}
				add_local_autonomy = 10
			}
			if = {
				limit = { can_build = paved_road }
				add_dip_power = -40
				add_building_construction = {
					building = paved_road
					speed = 0.5 # 50% of original speed
					cost = 0
				}
				add_local_autonomy = 10
			}
			if = {
				limit = { can_build = imperial_road }
				add_dip_power = -60
				add_building_construction = {
					building = imperial_road
					speed = 0.5 # 50% of original speed
					cost = 0
				}
				add_local_autonomy = 10
			}
		}
	}
	option = {
		name = "economy.10.b"
		ai_chance = { factor = 10 }
		
	}
}

########################################################################################
# import of horses
########################################################################################
country_event = {
	id = economy.12
	title = "economy.12.t"
	desc = "economy.12.d"
	picture = cavalry
	is_triggered_only = yes
	
	trigger = {
		NOT = {
			any_owned_province = { trade_goods = horses }
		}
		is_tribal = no
	}
	
	mean_time_to_happen = {
		days = 1
	}
	
	
	option = {
		name = "economy.12.a"
		add_years_of_income = -0.15
			add_country_modifier = {
				name = horses_imported
				duration = 1825
			}
	}
	option = {
		name = "economy.12.b"
		add_prestige = -2
			add_country_modifier = {
				name = no_horses_imported
				duration = 1825
			}
		
	}
}
