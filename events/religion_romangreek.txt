########################################################################################
# Giant Boar
########################################################################################
country_event = {
	id = religion_romangreek.1	
	title = "religion_romangreek.1.t"
	desc = "religion_romangreek.1.d"
	picture = boar
	is_triggered_only = yes

	trigger = {
		
		OR = {
		religion = roman_religion
		religion = greek_religion
		}
		
	}
	
	mean_time_to_happen = {
		days = 1
		
	}
	# kill the boars
	option = {
		name = "religion_romangreek.1.a"
			add_prestige = -5
			add_patriarch_authority = -0.03
	}
	# a sign of the gods
	option = {
		name = "religion_romangreek.1.b"
			add_prestige = 5
			random_owned_province = {
				add_province_modifier = {
					name = "boars"
					duration = 1095
				}
			}
			add_patriarch_authority = 0.03
				if = {
					limit = { has_global_flag = food_active }
						change_variable = { which = food_supplies value = -5 }
						while = {
							limit = { NOT = { check_variable = { which = food_supplies value = 0 } } }
							set_variable = { which = food_supplies value = 0 }
						}
					}
	}

}

	trigger = {
		num_of_cities = 3
		OR = {
		religion = roman_religion
		religion = greek_religion
		}
		
	}
	
	mean_time_to_happen = {
		days = 1
		
	}
	# to the capital
	option = {
		name = "religion_romangreek.2.a"
			add_patriarch_authority = 0.025
			define_advisor = {
				type = oracle
				skill = 1
				discount = yes
				female = yes
			}
			random_owned_province = {
				limit = { is_capital = no }
				add_province_modifier = {
					name = "oracle_removed"

				}
			}
	}
	# let her in peace
	option = {
		name = "religion_romangreek.2.b"

				limit = { is_capital = no }
				add_province_modifier = {
					name = "local_oracle"
					duration = 1825
				}
			}
	}

}

country_event = {
	id = religion_romangreek.8
	title = "religion_romangreek.8.t"
	desc = "religion_romangreek.8.d"
	picture = ritual_feast
	is_triggered_only = yes

	trigger = {
		OR = {
			primary_culture = umbri
			primary_culture = piceni
			primary_culture = samnitic
		}
		any_owned_province = { has_owner_culture = yes }
		is_tribal = yes
	}
	
	mean_time_to_happen = {
		days = 1
		
	}
	option = {
			name = "religion_romangreek.8.a" 
			random_owned_province = {
				limit = {	has_owner_culture = yes }
				add_patriarch_authority = 0.05
				add_base_manpower = 1
				add_province_modifier = {
					name = "sacred_spring_happened"
					duration = 1825
				}
			}
	}
	option = {
			name = "religion_romangreek.8.b" 
			add_patriarch_authority = -0.03
			add_prestige = -3
	}

}
