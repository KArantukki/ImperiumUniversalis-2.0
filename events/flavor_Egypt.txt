				}

				}

					factor = 1.5

				}

					factor = 0.75
					 num_of_loans = 1
				}

					factor = 0.5
					NOT = { manpower_percentage = 0.1 }
				}
				modifier = {
					factor = 0.5
					NOT = { stability = 0 }
				}
			}
				canaan_region = {
					limit = {
						NOT = { is_core = ROOT }
						NOT = { owned_by = ROOT }
					}
					add_claim = ROOT
				}
