########################################################################################
# no innovation
########################################################################################
country_event = {
	id = military.5
	title = "military.5.t"
	desc = "military.5.d"
	picture = training
	is_triggered_only = yes
	
	trigger = {
		army_professionalism = 0.2
		is_at_war = no
		NOT = { has_country_modifier = no_innovations_in_the_army }
		any_country = {
					OR = { 
						NOT = { border_distance = { who = ROOT distance = 500 } }
						is_neighbor_of = ROOT
					}
		}
	}
	
	mean_time_to_happen = {
		days = 1
	}

	immediate = {
		hidden_effect = {
			random_country = {
				limit = {
					OR = { 
						NOT = { border_distance = { who = ROOT distance = 500 } }
						is_neighbor_of = ROOT
					}
				}
				save_event_target_as = advisor_origin
			}
				set_saved_name = {
					key = hired_advisor_name
					type = advisor
					scope = event_target:advisor_origin
				}
		}
	}
	
	option = {
		name = "military.5.a"
		ai_chance = { factor = 40 }
			add_country_modifier = {
					name = no_innovations_in_the_army
					duration = 1825
				}
	}
	option = {
		name = "military.5.b"
		ai_chance = { factor = 20 }
			add_mil_power = -50
			add_army_tradition = -15
	}
	option = {
		name = "military.5.c"
		ai_chance = { factor = 40 }
			add_dip_power = -50
			add_years_of_income = -0.15
			define_advisor = {
				type = army_reformer_adv
				skill = 2
				name = hired_advisor_name
				culture = event_target:advisor_origin
				religion = event_target:advisor_origin
				discount = yes
			}
			
	}
	option = {
		name = "military.5.e"
		ai_chance = { factor = 100 }
		trigger = { ruler_has_personality = martial_educator_personality 	}
		required_personality = martial_educator_personality
		highlight = yes
		add_mil_power = -50
	}
	after = {
		clear_saved_name = hired_advisor_name
	}
	 
}

# revolting soldiers
########################################################################################
country_event = {
	id = military.7
	title = "military.7.t"
	desc = "military.7.d"
	picture = battle
	is_triggered_only = yes
	
	trigger = {
		always = no
		NOT = { land_maintenance = 0.75 }
		unrest = 1
		any_owned_province = { 
					num_of_units_in_province = {
						who = ROOT
						type = infantry
						amount = 5
					}
		}
	}
	
	mean_time_to_happen = {
		days = 1
		modifier = {
			factor = 1.5
			NOT = { land_maintenance = 0.45 }
		}
		modifier = {
			factor = 1.5
			unrest = 5
		}
	}
	
	
	immediate = { 
			hidden_effect = {
				random_owned_province = {
					limit = {
						num_of_units_in_province = {
							who = ROOT
							type = infantry
							amount = 5
						}
					}
				hidden_effect = {
						kill_units = {
							who = ROOT
							type = infantry
							amount = 5
						}
					}
					army_rebels = 5
				}
			}
	}
	

	option = {
		name = "military.7.a"
			tooltip = {
			random_owned_province = {
				limit = {
					num_of_units_in_province = {
						who = ROOT
						type = infantry
						amount = 5
					}
				}
				army_rebels = 5
			}
		}
	}
}
