	title = iu_institution_events.1.t
	desc = iu_institution_events.1.d
	picture = city
	is_triggered_only = yes

	trigger = {
		always = yes			#
	}

	mean_time_to_happen = {
		days = 1
	}

	option = {
		name = iu_institution_events.1.a
		add_adm_power = 100
		add_dip_power = 100
		add_mil_power = 100
		add_prestige = 10

			every_country = {
				limit = {
					ai = no
					NOT = { tag = ROOT }
				}
				country_event = { id = iu_institution_events.100 }
			}
		}
		
	}
}
