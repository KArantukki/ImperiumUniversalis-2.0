namespace = elections

# Election!
country_event = {
	id = elections.1
	title = "elections.1.t"
	desc = "elections.1.d"
	picture = senat
	
	is_triggered_only = yes
	
	trigger = {
		OR = {
		government = tribal_federation
		government = classic_republic
		government = punic_republic
		government = greek_polis_democracy
		government = league }
		NOT = {	government = monarchy }
		NOT = {	government = republican_dictatorship }
	}
	

	option = {
		name = "elections.a"	# An administrator type
		ai_chance = { factor = 30 }
		define_ruler = {
			adm = 2
			max_random_dip = 4
			max_random_mil = 4
			
		}
		add_adm_power = 20
		add_dip_power = -10
		add_mil_power = -10

	}
	option = {
		name = "elections.b"	# A diplomat
		ai_chance = { factor = 30 }
		define_ruler = {
			max_random_adm = 4
			DIP = 2
			max_random_mil = 4
			
		}
		add_adm_power = -10
		add_dip_power = 20
		add_mil_power = -10
	}
	option = {
		name = "elections.c"	# A military man
		ai_chance = { factor = 30 }
		define_ruler = {
			max_random_adm = 4
			max_random_dip = 4
			mil = 2					
			
		}
		add_adm_power = -10
		add_dip_power = -10
		add_mil_power = 20
	}
	option = {
		name = "elections.e"	# the old one
		ai_chance = { factor = 10 }
		
		if = {
			limit = { tag = ROM }
			add_republican_tradition = -15
		}
		if = {
			limit = { NOT = { tag = ROM } }
			add_republican_tradition = -10
		}
	}
}

# Election!Oligarchy
country_event = {
	id = elections.3
	title = "elections.3.t"
	desc = "elections.3.d"
	picture = senat
	
	is_triggered_only = yes
	trigger = {
		exists = yes
		NOT = { government = spartan_monarchy } 
	}
	

	option = {
		name = "elections.3.a"	# Oligarchic Type
		ai_chance = { factor = 35 }
		define_heir = {		
			age = 35
			hidden = yes
		}

		if = {
			limit = {
				has_dlc = "The Cossacks"
			}
			add_estate_loyalty = {
				estate = estate_nobles
				loyalty = 5
			}
			add_estate_loyalty = {
				estate = estate_citizens
				loyalty = 5
			}
		}
		if = {
			limit = {
					government = greek_polis
			}
			add_faction_influence = { faction = fac_oligarchs influence = 5 }
		}
		add_devotion = 5
		
	}
	option = {
		name = "elections.3.b"	# Democratic Type
		ai_chance = { factor = 25 }
		define_heir = {	
			age = 30
			hidden = yes
		}
		if = {
			limit = {
				has_dlc = "The Cossacks"
			}
			add_estate_loyalty = {
				estate = estate_nobles
				loyalty = -10
			}
			add_estate_loyalty = {
				estate = estate_citizens
				loyalty = 10
			}
		}
		if = {
			limit = {
					government = greek_polis
			}
			add_faction_influence = { faction = fac_democrats influence = 10 }
		}
		add_devotion = -5
		
	}
	option = {
		name = "elections.3.c"	# Tyrannic Type
		ai_chance = { factor = 30 }
		define_heir = {
			age = 20
			hidden = yes
		}
		if = {
			limit = {
				has_dlc = "The Cossacks"
			}
			add_estate_loyalty = {
				estate = estate_nobles
				loyalty = -5
			}
			add_estate_loyalty = {
				estate = estate_citizens
				loyalty = -10
			}
		}
		if = {
			limit = {
					government = greek_polis
			}
			add_faction_influence = { faction = fac_monarchists influence = 10 }
		}
		add_devotion = 10
		
	}
	
}
