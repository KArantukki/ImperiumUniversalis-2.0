				}
			}
        }
    
    mean_time_to_happen = {
		days = 1
           
    }
    option = {
        name = "tribal.17.a"
        ai_chance = {
			factor = 50 
			modifier = {
				factor = 0.1
				NOT = {
					estate_loyalty = {
						estate = estate_clans
						loyalty = 40
					}
				}
			}
		}
            add_estate_loyalty = {
				estate = estate_clans
				loyalty = -10
			}
    }
    option = {
        name = "tribal.17.b"
        ai_chance = { factor = 50 }

			random_owned_province = {
				limit = { 
					OR = {
						has_province_modifier = greek_trade_outpost
						has_province_modifier = phoenician_trade_outpost
						} 
					}
				remove_province_modifier = greek_trade_outpost
				remove_province_modifier = phoenician_trade_outpost
			}
    }
}
