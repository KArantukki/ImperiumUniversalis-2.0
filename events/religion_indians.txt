# buddhism appears
########################################################################################
country_event = {
	id = religion_indian.9
	title = "religion_indian.9.t"
	desc = "religion_indian.9.d"
	picture = priest
	major = yes
	
	trigger = {
		any_owned_province = {
			OR = {
				religion = brahmin
				religion = sramana
				religion = jain
			}
			OR = {
				area = magadha_area
				area = northern_kalinga_area
				area = gangaridai_area
				area = vrji_area
			}
		}
		is_year = 215
		NOT = { is_religion_enabled = buddhism }
		
	}
	
	mean_time_to_happen = {
		months = 60
		modifier = {
				factor = 1.5
				religion = jain
			}
	}
	
	option = {
		name = "religion_indian.9.a"
			enable_religion = buddhism
			random_owned_province = {
				limit = {
					OR = {
						area = magadha_area
						area = northern_kalinga_area
						area = gangaridai_area
						area = vrji_area
					}
					OR = {
						religion = sramana
						religion = brahmin
						religion = jain
					}
				}
				change_religion = buddhism
				add_reform_center = buddhism
				add_province_modifier = {
							name = home_of_buddhism
							duration = -1
						}
			}
	}
}
