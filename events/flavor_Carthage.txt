# trade outpost
########################################################################################
country_event = {
	id = carthage.9
	title = "carthage.9.t"
	desc = "carthage.9.d"
	picture = trade
	is_triggered_only = yes
	
	trigger = {
		OR = {
			primary_culture = phoenician
			primary_culture = punic
		}
		num_of_ports = 1
		any_country = {
			is_tribal = yes
			NOT = { culture_group = semitic }
			NOT = { border_distance = { who = ROOT  distance = 600 } }
			any_owned_province = { 
				has_port = yes
				NOT = { has_province_modifier = phoenician_trade_outpost }
				OR = {
					superregion = italia_superregion
					superregion = western_africa_superregion
					superregion = celtica_superregion
					region = thracia_region
					region = sarmatia_region
					}
				}
		}
	}
	
	mean_time_to_happen = {
		days = 1
		
		modifier = {
			factor = 2.0
			dip = 4
		}
		modifier = {
			factor = 2.0
			ruler_has_personality = entrepreneur_personality
		}
	}
	
	immediate = {
		random_country = {
			limit = {
				is_tribal = yes
				NOT = { culture_group = semitic }
				NOT = { border_distance = { who = ROOT  distance = 600 } }
				any_owned_province = { 
					has_port = yes
					NOT = { has_province_modifier = phoenician_trade_outpost }
					OR = {
						superregion = italia_superregion
						superregion = western_africa_superregion
						superregion = celtica_superregion
						region = thracia_region
						region = sarmatia_region
						}
					}
				}
			save_event_target_as = trade_parter_target
		}
	}
	
	option = {
		name = "greek.12.a"
		ai_chance = { factor = 90 }
		add_years_of_income = -0.1
		custom_tooltip = greek.12.tt
		hidden_effect = {
			event_target:trade_parter_target = {
				country_event = {
					id = carthage.10
					days = 90
				}
			}
		}
	}
	option = {
		name = "greek.12.b"
		highlight = yes
		trigger = {
			ruler_has_personality = entrepreneur_personality
		}
		required_personality = entrepreneur_personality
		ai_chance = { factor = 100 }
		add_dip_power = -25
		custom_tooltip = greek.12.tt
		hidden_effect = {
			event_target:trade_parter_target = {
				country_event = {
					id = carthage.10
					days = 90
				}
			}
		}
	}
	option = {
		name = "greek.12.c"
		ai_chance = { factor = 10 }
		add_prestige = -5
		add_mercantilism = -1
	}
}
