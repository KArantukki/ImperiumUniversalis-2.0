	desc = "greek.2.d"
	picture = senat
	fire_only_once = yes
	major = yes
	
	
	trigger = {
				is_year = 220.1.1
		NOT = { started_in = 244.1.1  }
		NOT = { has_global_flag = birth_of_democracy  }
		culture_group = hellenic
		government = greek_polis
		
		OR = {
			faction_influence = {
				faction = fac_democrats
				influence = 40
			}
			faction_in_power = fac_democrats
		}
			capital_scope = {
				OR = {
					region = greek_region
					region = mare_aegaeum_islands_region
					region = ionia_region
				}
			}
	}
	
	mean_time_to_happen = {
		months = 60
		modifier = {
				factor = 0.75
				TAG = ATH
			}
		modifier = {
				factor = 0.85
				NOT = { legitimacy = 65 }
			}
			
	}
	
	option = {
		name = "greek.2.a"
		set_global_flag = birth_of_democracy
		add_faction_influence = { faction = fac_democrats influence = 20 }
		add_legitimacy = 10
		every_country = {
				limit = {
					government = greek_polis
				}
				country_event = {
								id = greek.3
								days = 30
							}
			}
	}
}

country_event = {
	id = greek.6
	title = "greek.6.t"
	desc = "greek.6.d"
	picture = forum_picture
	major = yes
	
	trigger = {
		OR = {
				primary_culture = arcadian
				primary_culture = greek
			}
		capital_scope = { area = arcadia_area	}
		is_subject = no
		
		SP1 = {
			exists = yes
			total_development = 55
		}
		
		NOT = { exists = ARC}
		
		OR = {
			AND = {
				ai = yes
				NOT = {
					any_country = {
						ai = no
						OR = {
								primary_culture = arcadian
								primary_culture = greek
							}
						capital_scope = { region = greek_region	}
					}
				}
			}
				ai = no
		
		}
		
		calc_true_if = {
			all_country = {
				OR = {
				primary_culture = arcadian
				primary_culture = greek
				}
				capital_scope = { region = greek_region	}
				is_subject = no
				has_opinion = {
						who = ROOT
						value = -10
					}
			}
			amount = 2
		}
	}
	
	mean_time_to_happen = {
		months = 48
		modifier = {
			factor = 0.1
			SP1 = {
				total_development = 70 }
			}
		modifier = {
			factor = 0.4
			SP1 = {
				is_at_war = yes }
			}
		modifier = {
			factor = 4
			is_at_war = yes
		}
	}
	
	option = {
		name = "greek.6.a"
			hidden_effect = {
			every_country = {
					limit = {
					OR = {
						primary_culture = arcadian
						primary_culture = greek
						}
						capital_scope = { region = greek_region	}
						is_subject = no
						has_opinion = {
								who = ROOT
								value = -10
							}
					}
					country_event = {
						id = greek.7
						days = 1
					}
				}
			}
			custom_tooltip = greek.6.tt
			change_tag = ARC
			change_government = league
			if = {
				limit = { NOT = { government_rank = 2 } }
				set_government_rank = 2
			}
	}
	option = {
		name = "greek.6.b"
			add_stability = 1
	}
}
