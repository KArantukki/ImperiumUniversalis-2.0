age_of_empires = {
	start = 145
	
	can_start = {
		always = yes
	}
	
	religious_conflicts = yes
	
	absolutism = {
		stability = 1
		strengthen_government = 2
		
		debase_currency = -1
		war_exhaustion = -1
	}

	objectives = {	
	
		obj_great_power = {
			allow = { always = yes }
				NOT = { great_power_rank = 5 }
		}	
	
		obj_legendary_ruler = {
			allow = { always = yes }
			ADM = 5
			DIP = 5
			MIL = 5
		}
	
		obj_extensive_country = {
			allow = { always = yes }
			num_of_states = 10
		}
	
		obj_exploration_of_the_world_west = {
			allow = {
				NOT = { capital_scope = { superregion = iran_superregion } }
				NOT = { capital_scope = { continent = india } }
				NOT = { capital_scope = { continent = arabia_superregion } }
			}
			 has_discovered = 1337
			 has_discovered = 612
			 has_discovered = 10
			 has_discovered = 959
			 has_discovered = 745
			 has_discovered = 2476
			 has_discovered = 2500
		}
	
		obj_exploration_of_the_world_east = {
			allow = {
				OR = { 	capital_scope = { superregion = iran_superregion } 
						capital_scope = { continent = india } 
						capital_scope = { continent = arabia_superregion }  }
			}
			 has_discovered = 2682
			 has_discovered = 2864
			 has_discovered = 2930
			 has_discovered = 3027
			 has_discovered = 2476
			 has_discovered = 2500
		}
	
		obj_unified_culture_group = {
			allow = { always = yes }
			custom_trigger_tooltip = {
				tooltip = obj_unified_culture_group.tt
			has_unified_culture_group = yes 
			}
		}
	
		obj_advanced_horsebreeding = {
			allow = { always = yes }
			 full_idea_group = cavalry_ideas 
			 any_owned_province = {
				trade_goods = horses
			 }
		}
		
		obj_coinage = {
	 		has_institution = coin_usage
		}
	
		
		
	}	

	abilities = {
	
		ab_chariotry = {
			modifier = {
				cavalry_power = 0.1 
			}
			ai_will_do = {
				factor = 10
			}			
		}
	
		ab_restless_conquerors = {
			modifier = {
				core_creation = -0.2
			}
			ai_will_do = {
				factor = 10
			}			
		}
	
		ab_new_trade_routes = {
			modifier = {
				global_trade_power = 0.1 
			}
			ai_will_do = {
				factor = 10
			}			
		}
	
		ab_new_alliances = {
			modifier = {
				envoy_travel_time = -0.25
				diplomats = 2
			}
			ai_will_do = {
				factor = 10
			}			
		}
	
		ab_ocean_goining_ships = {
			modifier = {
				ship_durability = 0.1
				trade_range = 1.0
				range = 1.0
			}
			ai_will_do = {
				factor = 10
			}			
		}
	
		ab_unquestioned_overlordship = {
			modifier = {
				reduced_liberty_desire = 25
			}
			ai_will_do = {
				factor = 10
			}
		}
	
		ab_stabilize_the_country = {
			effect = {
				add_stability = 2
				add_war_exhaustion = -2
				add_corruption = -2 
			}
			ai_will_do = {
				factor = 5
			}
		}
	
		ab_babylonian_empire = {
			allow = {
				tag = BAB
			}
			modifier = {
				build_cost = -0.15
			}
			ai_will_do = {
				factor = 50
			}			
		}
	
		ab_median_horsemenship = {
			allow = {
				tag = MDI
			}
			modifier = {
				cavalry_flanking = 0.33
			}
			ai_will_do = {
				factor = 50
			}			
		}
	
		ab_lydian_coinage = {
			allow = {
				tag = LYD
			}
			modifier = {
				inflation_reduction = 0.2
				global_tax_modifier = 0.1
			}
			ai_will_do = {
				factor = 50
			}			
		}
	
		ab_scythian_fury = {
			allow = {
				OR = {
					primary_culture = scythian
					primary_culture = cimmerian
				}
			}
			modifier = {
				fire_damage = 0.1 
				shock_damage = 0.1 
			}
			ai_will_do = {
				factor = 50
			}			
		}
		
		
	}
}
