	reinforcing = yes				# Can reinforce

			factor = 3

				estate_influence = { estate = estate_heilotes influence = 80 }
			}
		}
		modifier = {
			factor = 3
			has_estate = estate_heilotes
		}
		modifier = {
			factor = 0.1
				owner = { 
					NOT = { check_variable = { which = slaves value = 500 } }

			factor = 2

		modifier = {
