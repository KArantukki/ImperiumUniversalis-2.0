# When someone is privateering in a node that you trade and have some power in.

	prerequisites = {
		is_free_or_tributary_trigger = yes
		FROM = {
			has_privateers = yes
		}
		or = {
			home_trade_node = {
				trade_share = {
					country = ROOT
					share = 10
				}
				privateer_power = {
					country = FROM
					share = 1
				}
			}
			any_active_trade_node = {
				trade_share = {
					country = ROOT
					share = 10
				}
				privateer_power = {
					country = FROM
					share = 1
				}
			}
		}
		is_revolution_target = no
	}

	war_goal = blockade_ports
}
