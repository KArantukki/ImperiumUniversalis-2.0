writing_system = {
	penalty = 0.5
	bonus = {
		advisor_cost = -0.1
		advisor_pool = 1
	}
	historical_start_date = 100.1.1
	historical_start_province = 1813

	history = {
		OR = {
				owned_by = TRQ
				culture_group = hellenic
				culture_group = anatolian_group
				culture_group = semitic
				culture_group = caanan_group
				culture_group = akkadian_group
				culture_group = western_iranian

				region = italian_region
				region = magna_graecia
			AND = {
				culture_group = turdetani_group
				is_year = 210
			}
			AND = {
				culture_group = iberian_group
				is_year = 250
			}
			AND = {
				culture_group = eastern_iranian_group
			}
			AND = {
				culture_group = armenian_group

			}
			AND = {

			is_tribal = no

		}
		
		
	}
	
	# no need for events, as its there from the start.
	#can_start = {	is_year = 1000	}
	#start_chance = 100
	#on_start = institution_events.1

	can_embrace = {
		always = yes
	}

	embracement_speed = {
		
		modifier = {
			scale = yes
			factor = 1.0
			custom_trigger_tooltip = {
				tooltip = tooltip_any_friendly_coast_border_province_has_embraced_writing_system
				any_friendly_coast_border_province = {
					writing_system = 100
				}
			}
		}
		modifier = {
			scale = yes
			factor = 0.5
			potential = {
				NOT = {
					any_friendly_coast_border_province = {
						writing_system = 100
					}
				}
			}
			custom_trigger_tooltip = {
				tooltip = tooltip_any_border_province_has_embraced_writing_system
				any_neighbor_province = {
					NOT = { owned_by = ROOT }
					writing_system = 100
				}
			}
		}
		modifier = {
			scale = yes
			factor = 0.5
			potential = {
			OR = {
					has_province_modifier = center_of_trade_modifier_1
					has_province_modifier = center_of_trade_modifier_2
					has_province_modifier = center_of_trade_modifier_3
					has_province_modifier = center_of_trade_silkroad_modifier
				}
			}
			custom_trigger_tooltip = {
				tooltip = tooltip_any_center_of_trade_has_embraced_writing_system
				OR = {
					has_province_modifier = center_of_trade_modifier_1
					has_province_modifier = center_of_trade_modifier_2
					has_province_modifier = center_of_trade_modifier_3
					has_province_modifier = center_of_trade_silkroad_modifier
				}
				any_trade_node_member_province = { 
					writing_system = 100
				}
			}
		}
		modifier = {
			scale = yes
			factor = 0.5
			custom_trigger_tooltip = {
				tooltip = tooltip_has_upper_class
				base_tax = 1
			}
		}
		modifier = {
			scale = yes
			factor = 0.25
			potential = {
				is_capital = yes
				owner = { is_tribal = no }
			}
			custom_trigger_tooltip = {
				tooltip = skilled_ruler_adm
				owner = {
					ADM = 6
				}
			}
		}
		modifier = {
			scale = yes
			factor = 0.5
			potential = {
				is_capital = yes
				owner = { is_tribal = no }
			}
			custom_trigger_tooltip = {
				tooltip = has_idea_group_scholars_ideas
				owner = {
					full_idea_group = scholars_ideas
				}
			}
		}
		modifier = {
			scale = yes
			factor = 0.25
			potential = {
				is_capital = yes
				owner = { is_tribal = no }
			}
			custom_trigger_tooltip = {
				tooltip = has_advisor_scholar_skill
				owner = {
					OR  = {
							scholar_advisor = 1
							scholar_advisor = 2
							scholar_advisor = 3
						}
				}
			}
		}
	}
	
	ai_will_do = { #Cost of embracing institution must be lower than this factor * last month's regular income. (Or it has a more than long term goal...)
		factor = 24
		
		modifier = {
			factor = 0.5
			is_at_war = yes
		}
	}
}
