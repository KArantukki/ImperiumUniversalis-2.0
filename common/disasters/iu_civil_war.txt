iucivilwar = {
	potential = {
		num_of_cities = 5
		NOT = { has_any_disaster = yes }
		NOT = { government = roman_senate }
		is_subject = no
		is_tribal = no
		government_rank = 2
	}

	
	
	can_start = {
		has_any_disaster = no
		OR = {
			AND = {		government = republic
				NOT = { republican_tradition = 50 } }
			AND = {		government = monarchy
				NOT = { legitimacy = 50 } }	
			AND = {		government = oligarchic_monarchy
				NOT = { devotion = 50 } }
			}	
		}
	
	
	can_stop = {
		OR = {
			AND = {		government = republic
						republican_tradition = 50 
						}
			AND = {		government = monarchy
						legitimacy = 50
						}
			AND = {		government = oligarchic_monarchy
						devotion = 50
						}
			AND = {		government = republican_dictatorship
						stability = 1 }	
			}
		}
		
	
	
	progress = {
		
		modifier = {
			factor = 1
			OR = {
			AND = {		government = republic
				NOT = { republican_tradition = 50 } }
			AND = {		government = monarchy
				NOT = { legitimacy = 50 } }	
			}
			AND = {		government = oligarchic_monarchy
				NOT = { devotion = 50 }  }
		}
		modifier = {
			factor = 1
			OR = {
			AND = {		government = republic
				NOT = { republican_tradition = 40 } }
			AND = {		government = monarchy
				NOT = { legitimacy = 40 } }	
			}
			AND = {		government = oligarchic_monarchy
				NOT = { legitimacy = 40 } }	
			}
		modifier = {
			factor = 1
			overextension_percentage = 0.10
		}		
		modifier = {
			factor = 2
			revolt_percentage = 0.10
		}
		modifier = {
			factor = 2
			NOT = { stability = 0 }
		}
		modifier = {
			factor = 2
			NOT = { stability = -1 }
		}
		modifier = {
			factor = 2
			NOT = { stability = -2 }
		}		
	}
	
	
	
	can_end = {
		OR = {
			AND = {		government = republic
						republican_tradition = 90 }
			AND = {		government = monarchy
						legitimacy = 90 }
			AND = {		government = republican_dictatorship
						republican_tradition = 90 }
			AND = {		government = oligarchic_monarchy
						devotion = 90 }
		}
						stability = 0
	}
	

	modifier = {
		manpower_recovery_speed = -0.25
		global_autonomy = 0.05
		global_unrest = 5
	}	
	
	
	on_start = iucivilwar.1
	on_end = iucivilwar.2
	
	on_monthly = {
		events = {
		}
		random_events = { 
			1000 = 0
			100 = iucivilwar.3
			100 = iucivilwar.4
			100 = iucivilwar.5
			100 = iucivilwar.6
			100 = iucivilwar.7
			100 = iucivilwar.8
		}
	}
}
