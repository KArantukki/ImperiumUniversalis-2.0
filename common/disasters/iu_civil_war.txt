iucivilwar = {
	potential = {
		num_of_cities = 5
		NOT = { has_any_disaster = yes }
		#NOT = { government = roman_senate }
		is_subject = no
		is_tribal = no
		NOT = { government = native }
		government_rank = 2
	}

	
	
	can_start = {
		NOT = { stability = 0 }
		NOT = { manpower_percentage = 0.50 }
		}
	
	
	can_stop = {
			stability = 1
			manpower_percentage = 0.50 
		}
		
	
	
	progress = {
		modifier = {
			factor = 1
			NOT = { stability = 0 }
		}
		modifier = {
			factor = 1
			NOT = { stability = -1 }
		}
		modifier = {
			factor = 1
			NOT = { stability = -2 }
		}
		modifier = {
			factor = 1
			NOT = { manpower_percentage = 0.50 }
			}
		modifier = {
			factor = 1
			NOT = { manpower_percentage = 0.30 }
			}
		modifier = {
			factor = 1
			NOT = { manpower_percentage = 0.10 }
			}
	}
	
	
	
	can_end = {
		stability = 1
		manpower_percentage = 0.50 
	}
	

	modifier = {
		stability_cost_modifier = 0.5
		legitimacy = -1
		republican_tradition = -0.5
		devotion = -0.5
		global_unrest = 5
		
	}	
	
	
	on_start = iucivilwar.1
	on_end = iucivilwar.2
	
	on_monthly = {
		events = {
		}
		random_events = { 
			1000 = 0

		}
	}
