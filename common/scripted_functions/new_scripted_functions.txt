can_improve_tax = {
	condition = {
		tooltip = "NEEDRANK_2"
		potential = {
			owner = { 
				NOT = { government_rank = 3 }
			}
		}
		allow = {
			always = no
		}
	}
	condition = {
		tooltip = "IS_STATE_COL"
		potential = {
			is_state = no
		}
		allow = {
			always = no
		}
	}
	condition = {
		tooltip = "NOT_POSSIBLE"
		potential = {
			owner = { 
				has_country_flag = colony_event_flag
			}
		}
		allow = {
			always = no
		}
	}
	condition = {
		tooltip = "IS_TRIBAL_OWNER"
		potential = {
			owner = { 
				OR = {
					is_tribal = yes
					is_nomad = yes
				}
			}
		}
		allow = {
			always = no
		}
	}
	condition = {
		tooltip = "DIFFERENT_CULTURE"
		potential = {
			has_owner_culture = yes
		}
		allow = {
			has_owner_culture = no
		}
	}
	condition = {
		tooltip = "NOT_AI"
		potential = {
			owner = { ai = yes }
		}
		allow = {
			ai = no
		}
	}
	condition = {
		tooltip = "ALREADY_TARGET_OF_COLONY"
		potential = {
			OR = {
				has_province_modifier = building_colony
			}
		}
		allow = {
			NOT = { has_province_modifier = building_colony }
		}
	}
	condition = {
		tooltip = "COLONY_IN_SUBJECT_PROV"
		potential = {
			owner = { is_subject_of = ROOT }
		}
		allow = {
			always = no
		}
	}
	condition = {
		tooltip = "NEEDS_POPULATION"
		potential = {
			NOT = { development = 1 }
		}
		allow = {
			always = no
		}
	}
}
